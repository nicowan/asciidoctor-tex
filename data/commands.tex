% Defines the adocEnvQuote environment
% Takes 2 parameters : <Title> and <Attribution>
\newcommand{\adocQuoteAttribution}{~}
\newenvironment{adocEnvQuote}[2]{%
  \renewcommand{\adocQuoteAttribution}{#2}%

  \begin{tcolorbox}[title={#1}, boxrule=0, opacityframe=0.25, colback=white, lefthand width=1cm, sidebyside, sidebyside gap=0mm,  sidebyside align=top seam, lower separated=false]
    \includegraphics[height=1em]{open-quotation}%
    \tcblower
}{%

    \ifx\adocQuoteAttribution\empty \else
      \hfill\textit{\adocQuoteAttribution}
    \fi
    \end{tcolorbox}
}

% Defines the adocEnvVerse environment
